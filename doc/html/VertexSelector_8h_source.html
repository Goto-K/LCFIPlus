150  // true if vtx is V0
151  bool isV0(const Vertex* vtx, const VertexSelectorConfig& cfg, const Vertex* primary) {
152  // two-track selection
153  if ( vtx->getTracks().size() !=2 )
154  return false;
155 
156  // charge selection
157  if ( vtx->getTracks()[0]->getCharge() * vtx->getTracks()[1]->getCharge() != -1 )
158  return false;
159 
160  // obtain momentum vectors at the V0 vertex (IMPORTANT)
161  TVector3 mom1 = vtx->getTracks()[0]->momentumAtVertex(vtx);
162  TVector3 mom2 = vtx->getTracks()[1]->momentumAtVertex(vtx);
163 
164  // V0 direction selection
165  TVector3 posip;
166  if (primary) posip = primary->getPos();
167  double v0dirdot = (mom1+mom2).Unit().Dot( (vtx->getPos()-posip).Unit() );
168 
169  // K0 selection
170  if (cfg.rejectk0) {
171  TLorentzVector lvec1;
172  TLorentzVector lvec2;
173  lvec1.SetVectM( mom1, 0.1396 );
174  lvec2.SetVectM( mom2, 0.1396 );
175  double mass = (lvec1+lvec2).M();
176  double k0mass = .498;
177  if (v0dirdot > cfg.k0dirdot && fabs(mass-k0mass) < cfg.k0width )
178  return true;
179  }
180 
181  // Lambda selection
182  if (cfg.rejectl0) {
183  // assume higher momentum particle is the proton
184  TLorentzVector protonForLambda;
185  TLorentzVector pionForLambda;
186  if (mom1.Mag() > mom2.Mag()) {
187  protonForLambda.SetVectM( mom1, 0.9383 );
188  pionForLambda.SetVectM( mom2, 0.1396 );
189  } else {
190  protonForLambda.SetVectM( mom2, 0.9383 );
191  pionForLambda.SetVectM( mom1, 0.1396 );
192  }
193 
194  double lambdaMass = (protonForLambda+pionForLambda).M();
195  if (v0dirdot > cfg.l0dirdot && fabs(lambdaMass - 1.1157) < cfg.l0width)
196  return true;
197  }
198 
199  // check photon conversion; photon mass corrected geometrically
200  if (cfg.rejectconv) {
201  double ang1 = atan( vtx->getTracks()[0]->getTanLambda());
202  double ang2 = atan( vtx->getTracks()[1]->getTanLambda());
203  double convmassCorr = sqrt( vtx->getTracks()[0]->Vect().Mag()*vtx->getTracks()[1]->Vect().Mag()*(1-cos(ang1-ang2)) );
204  if (v0dirdot > cfg.convdirdot && vtx->getPos().Mag()> cfg.convdist && convmassCorr < cfg.convmass)
205  return true;
206  }
207 
208  return false;
209  }
